def initialize(_user = nil, experiment = nil, variation = -1, forced = false)
  @experiment = experiment
  @variation = variation
  @forced = forced

  @data = {}
  return unless experiment&.data

  var = [variation, 0].max
  experiment.data.each do |k, v|
    @data[k] = v[var]
  end
end
