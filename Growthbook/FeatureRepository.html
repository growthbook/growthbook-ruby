def fetch
  uri = URI(endpoint)
  res = Net::HTTP.get_response(uri)

  @response = res.is_a?(Net::HTTPSuccess) ? res.body : nil

  return nil if response.nil?

  if use_decryption?
    parsed_decrypted_response
  else
    parsed_plain_text_response
  end
rescue StandardError
  nil
end
